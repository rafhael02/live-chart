function spawnEmojis(boxId, emoji) { const count = Math.floor(Math.random() * 3) + 3; // 3-5 emojis const box = document.getElementById(boxId); const rect = box.getBoundingClientRect(); for (let i = 0; i < count; i++) { const el = document.createElement("div"); el.className = "emoji"; el.textContent = emoji; // spawn anywhere inside the box width const x = rect.left + Math.random() * rect.width; const y = rect.top + window.scrollY; el.style.left = x + "px"; el.style.top = y + "px"; // vary animation duration for natural look el.style.animationDuration = (1.5 + Math.random() * 1) + "s"; document.body.appendChild(el); setTimeout(() => el.remove(), 2000); } }
